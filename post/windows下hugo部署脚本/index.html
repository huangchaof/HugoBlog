@echo off

echo "Enter blog  Dir E:\HugoBlog\public "
echo "***********************************"
cd /d  E:\HugoBlog\public 

echo "Repo status "
echo "***********************************"

Rem git status | grep "working tree clean" 如果匹配正确 ，ERRORLEVEL的值为1，否则为0
git status | grep "working tree clean"

Rem echo %ERRORLEVEL%  打印值  
if %ERRORLEVEL% == 1 (goto fun1 )
if %ERRORLEVEL% == 0 (goto fun2)

Rem  不能把fun1 fun2标签段理解为函数，因为如果前面的执行没有退出，后面会按顺序执行fun2和fun1
Rem  goto:eof  就是执行到脚本末尾，不再顺序执行脚本
:fun2
	echo "no changes press any key to exit"
	pause
Goto:eof

:fun1
	echo "Add all changes "
	echo "***********************************"
	git add . 
	echo "Commit changes "
	echo "***********************************"
	git commit -m "new post"
	echo "Pull remote repo " 
	echo "***********************************"
	git pull  
	echo "Push to remote repo"
	echo "***********************************"
	git push
	echo  "End press any key to exit"
	pause 
Goto :eof
